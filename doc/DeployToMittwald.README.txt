How to deploy an extension to Mittwald using git:

1. Log onto Account
	ssh pXXXXX@pXXXXX.mittwaldserver.info
2. initilize bare git repo in files/
	cd files
	mkdir snet_tmpl_basic.git
	cd snet_tmpl_basic.git
	git init --bare
3. ON LOCAL MASCHINE. Push to Mittwald
	cd []/snet_tmpl_basic
	git remote add Mittwald pXXXXX@pXXXXX.mittwaldserver.info:files/snet_tmpl_basic.git
	git push Mittwald +master:refs/heads/master
	git push Mittwald
4. ON MITTWALD clone bare repo into docroot
	cd /html/typo3/typo3conf/ext
	git clone /files/snet_tmpl_basic.git
5. prepare post update hook to deploy to docroot
	cd /files/snet_tmpl_basic.git/hooks/
	echo "GIT_WORK_TREE=/html/typo3/typo3conf/ext/snet_tmpl_basic  git checkout -f" > post-update
	chmod +x post-update
